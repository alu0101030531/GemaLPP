  while node != nil do
    yield node[:value]
    node = node[:next]
  end

    Inserta en la lista un nodo en la última posición

    # File lib/FoodImpact/list.rb, line 22
def insert(object)
  node = Node.new(object, nil, nil)
  if @sz_ == 0
    @head_ = node
    @tail_ = node
  else
    @tail_[:next] = node
    node[:prev] = @tail_
    @tail_ = node
  end
  @sz_+=1
end
