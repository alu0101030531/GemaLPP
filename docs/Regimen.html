def gei
  food = @ingredientes.head
  gases = 0.0
  while food != nil
    gases += food[:value].gei * (food[:value].gramos / 1000.0)
    food = food[:next]
  end
  gases
end

def huella
  if terreno < 800
    terreno_index = 1
  elsif terreno > 1200
    terreno_index = 3
  else
    terreno_index = 2
  end
  if kcalorias < 670
    energia_index = 1
  elsif kcalorias > 830
    energia_index = 3
  else
    energia_index = 2
  end
  (terreno_index + energia_index)/ 2
end

def terreno
  food = @ingredientes.head
  terreno = 0.0
  while food != nil
    terreno += food[:value].terreno * (food[:value].gramos)
    food = food[:next]
  end
  terreno
end
